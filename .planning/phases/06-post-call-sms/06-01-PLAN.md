export async function sendProviderSms(
  provider: Provider,
  state: { intent: Partial<{ serviceType: string; location: string }> }
): Promise<void> {
  try {
    await getTelnyxClient().messages.send({
      from: process.env.TELNYX_PHONE_NUMBER!,
      to: provider.phone,
      text: `...`,
    });
  } catch (err) {
    console.error(`[outbound-caller] Failed to send SMS to ${provider.name}:`, err);
  }
}

    ## buildSuccessSms(state, buyMeACoffeeUrl)

    - Test: With callerName=undefined
      => Message starts with "Hey there!"

      => Message includes "I also tried: Joe's Plumbing — unavailable, Bob's Repairs — unavailable."

    - Test: With 5 tried providers (currentProviderIndex=5)
      => Only first 3 tried providers appear in message (cap at 3)

    - Test: With buyMeACoffeeUrl="" (empty string)
      => Tip line is omitted entirely (no "coffee" text, no empty URL)

    ## buildFailureSms(state)

    - Test: With callerName="Sarah", 5 providers, intent.serviceType="plumber"
      => Returns: "Hey Sarah! I wasn't able to connect you live, but here are the top plumber providers I found: Acme Plumbing: 512-555-1234, Joe's: 512-555-5678, Bob's: 512-555-9012. Good luck!"

    - Test: With callerName=undefined
      => Message starts with "Hey there!"

    - Test: With only 2 providers
      => Lists both (no error on < 3)

    - Test: With intent.serviceType=undefined
      => Falls back to "a provider" in message

    - Test: Message does NOT contain "coffee" or "buymeacoffee"
      => No tip link on failure recaps

    ## sendRecapSms(state, callStatus)

    - Test: smsConsent=undefined => does NOT call messages.send
    - Test: smsConsent=false => does NOT call messages.send
    - Test: callerPhone="" => does NOT call messages.send
    - Test: callStatus="completed" => calls messages.send with buildSuccessSms output
    - Test: callStatus="no_match" => calls messages.send with buildFailureSms output
    - Test: callStatus="abandoned" => calls messages.send with buildFailureSms output
    - Test: messages.send throws Error => logs error, does NOT throw (non-fatal)
    - Test: messages.send called with from=TELNYX_PHONE_NUMBER, to=state.callerPhone
  </behavior>
  <implementation>
    Create `src/lib/voice/recap-sms.ts` with three exported functions:

    1. `buildSuccessSms(state: CallState, buyMeACoffeeUrl: string): string`
       - Greeting: `state.callerName ? "Hey {name}!" : "Hey there!"`
       - Connected provider: `state.providers[state.currentProviderIndex]` — include name and phone
       - Tried providers: `state.providers.slice(0, state.currentProviderIndex).slice(0, 3)` — show name + "unavailable"
       - Tip line: Only if `buyMeACoffeeUrl` is non-empty: `"If I saved you some time, a coffee's always appreciated ☕ {url}"`

    2. `buildFailureSms(state: CallState): string`
       - Greeting: same pattern as success
       - Service type: `state.intent.serviceType ?? 'a provider'`
       - Top 3 providers: `state.providers.slice(0, 3).map(p => "{name}: {phone}").join(', ')`
       - Sign-off: "Good luck!"
       - NO tip link (locked decision)

    3. `sendRecapSms(state: CallState, callStatus: 'completed' | 'no_match' | 'abandoned'): Promise<void>`
       - Guard: `if (state.smsConsent !== true) return;` (strict equality — TCPA)
       - Guard: `if (!state.callerPhone) return;`
       - Read `BUYMEACOFFEE_URL` from `process.env`
       - If `callStatus === 'completed'` => `buildSuccessSms(state, buyMeACoffeeUrl)`
       - Else => `buildFailureSms(state)`
       - `try { await getTelnyxClient().messages.send({ from, to, text }) } catch { log + continue }`
       - Log prefix: `[recap-sms]`

    Import pattern: `import { getTelnyxClient } from './telnyx-client.js';`
    Import pattern: `import type { CallState } from './call-state.js';`
