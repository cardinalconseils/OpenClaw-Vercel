1. Create `frontend/src/components/dashboard/profile-form.tsx` — client component ('use client'):
   - Props: `{ user: { email: string; fullName?: string; phone?: string } }`
   - Zod schema for validation:
     ```typescript
     import { z } from 'zod'
     const profileSchema = z.object({
       fullName: z.string().min(1, 'Name is required').max(100, 'Name too long'),
       phone: z.string().regex(/^\+?[1-9]\d{1,14}$/, 'Enter a valid phone number (E.164 format)').or(z.literal('')),
       email: z.string().email('Enter a valid email'),
     })
     ```
   - Use `react-hook-form` with `zodResolver`:
     ```typescript
     import { useForm } from 'react-hook-form'
     import { zodResolver } from '@hookform/resolvers/zod'
     const form = useForm<z.infer<typeof profileSchema>>({
       resolver: zodResolver(profileSchema),
       defaultValues: { fullName: user.fullName ?? '', phone: user.phone ?? '', email: user.email },
     })
     ```
   - Form fields using shadcn Form components: FormField > FormItem > FormLabel > FormControl > Input > FormMessage
   - Fields: Full Name (text), Phone Number (tel), Email (email, disabled — Supabase handles email changes separately)
   - Submit handler:
     ```typescript
     async function onSubmit(values: z.infer<typeof profileSchema>) {
       const supabase = createClient()
       const { error } = await supabase.auth.updateUser({
         data: { full_name: values.fullName, phone: values.phone }
       })
       if (error) { setFormError(error.message) }
       else { setSuccess(true); setTimeout(() => setSuccess(false), 3000) }
     }
     ```
   - Submit button: "Save Changes" primary variant, disabled during submission (show loading spinner via `form.formState.isSubmitting`)
   - Success message: green text "Profile updated successfully" below form
   - Error message: destructive text below form (per UI-SPEC: inline, not toast)
